# 探索 Webpack 的打包实现原理

通过对 Webpack 打包出来的 bundle 文件分析，我们了解了 Webpack 打包后的结果是如何怎样运行起来的。我们再来想一想，到底如何可以实现一个与 Webpack 功能类似的bundler 打包器呢？

# 分析完成打包需要做的事情

  1. 实现是一个模块解析方法，记录各个文件的依赖关系并完成 ES6 等新语法到 ES5 的转换，这一步我们可以借助Babel工具来完成：
  • 使用@babel/parser 将代码转换为AST抽象语法树
  • 利用@babel/traverse 对 AST 进行遍历，定位到模块导入语句并记录下依赖关系
  • 通过@babel/core 和 @babel/preset-env 进行语法的转换

写一个 parseModule 方法

const fs = require("fs");
const path = require("path");
const babel = require("@babel/core");
const babylon = require("@babel/parser");
const traverse = require("@babel/traverse").default;

//解析模块方法: 接受一个模块的绝对路径，解析、转换模块代码，返回模块信息
function parseModule(filename) {
  //使用fs模块读取文件信息
  const fileContent = fs.readFileSync(filename, "utf-8");
  const ast = babylon.parse(fileContent, {
    sourceType: "module",
  });

  const dependencies = {}; //创建一个对象来来存放当前模块的依赖关系
  //遍历抽象语法树
  traverse(ast, {
    //定位到 Import 导入语句
    ImportDeclaration({ node }) {
      const dirname = path.dirname(filename);
      const fileRelativePath = "./" + path.join(dirname, node.source.value);
      dependencies[node.source.value] = fileRelativePath;
    },
  });

  //进行代码转换
  const { code } = babel.transformFromAst(ast, null, {
    presets: ["@babel/preset-env"],
  });
  //返回当前模块的依赖信息和转换后的代码
  return {
    filename,
    dependencies,
    code,
  };
}

遍历抽象语法树定位到具体的语句这一部分其实上次广爷已经给我们讲过了，配合这个超好用的语法实时转换AST的工具网站，整个过程其实还是非常容易的。

  1. 第二步，实现一个getModuleList方法用来获取完整的模块列表, 得到每一个模块的ID 和模块代码的映射关系
//创建完整的模块列表
function getModuleList(entry) {
  //从入口模块开始解析
  const entryModule = parseModule(entry);

  //创建一个模块信息列表，用于存放所有的模块信息
  const moduleInfoArray = [entryModule];
  //遍历moduleInfoArray开始解析所有模块
  for (let i = 0; i < moduleInfoArray.length; i++) {
    const item = moduleInfoArray[i];
    const { dependencies } = item; //拿到文件所依赖的模块集合
    for (let j in dependencies) {
      //这里的目的是将入口模块及其所有依赖都存入数组
      let module = parseModule(dependencies[j]);
      moduleInfoArray.push(module);
    }
  }

  //开始生成模块列表
  const moduleList = {};
  moduleInfoArray.forEach((item) => {
    moduleList[item.filename] = {
      dependencies: item.dependencies,
      code: item.code,
    };
  });

  return moduleList;
}
  1. 最终步骤：生成 bundle.js
function createBundle(entry) {
  //将入口信息传入getModuleList，得到模块的映射谱图
  const modules = getModuleList(entry);
  const bundle = `
    (function(modules) {
        //require函数的本质是执行一个模块的代码，然后将相应变量挂载到exports对象上
        function require(module) {
            //localRequire的本质是拿到依赖包的exports变量
            function localRequire(relativePath) {
              return require(modules[module].dependencies[relativePath]);
            }
            var exports = {};
            (function(require, exports, code) {
              eval(code);
            })(localRequire, exports, modules[module].code);
            return exports;//函数返回指向局部变量，形成闭包，exports变量在函数执行后不会被摧毁
        }
        require('${entry}')
      })(${JSON.stringify(modules)})
    `;
  //将最终结果写入bundle.js文件中
  fs.writeFileSync("./dist/bundle.js", bundle);
}

其实使用babel进行代码转换后，我们使用的ESM 语法被转换成了require。我们需要在浏览器实现一个require函数才能够正常运行。

# 说一说require的原理

bundle在浏览器运行的核心其实就是实现了一个require，我们再顺便提一下commonJs里面require函数的实现原理吧，理解一下require的本质到底是什么。

node 中的 require核心源码

const fs = require('fs');
const path = require('path');
const vm = require('vm'); // 虚拟机，提供沙箱环境
const wrapper = [
    '(function(exports, module, require){',
    '})'
]
function Module(absPath) {
    this.id = absPath; // 存储当前路径
    this.exports = {};
}

Module.prototype.load = function() {
    let script = fs.readFileSync(this.id, 'utf8');
    let fnStr = wrapper[0] + script + wrapper[1];
    let fn = vm.runInThisContext(fnStr); // 沙箱执行当前函数
    fn(this.exports, this, req); // 让拼出的函数执行
}

function my_require(file ) {
   let absPath = path.resolve(__dirname, file); // 1. 获取文件路径
   let module = new Module(absPath); // 2. 创建一个模块
   module.load(); // 3. 加载方法
   return module.exports;
}
